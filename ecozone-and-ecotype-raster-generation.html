SAR and Ecotype Project

9 Ecozone and Ecotype Raster Generation

This document outlines the workflow to generate ecozone- and ecotype-specific raster layers representing ecotype scores across Canada.

9.1 Overview: Ecozone Loop

The ecozone loop produces ecozone-specific raster layers where each 30m ecotype cell is reclassified to its unique score for that ecozone.

The workflow involves:

  1. Masking the NALCMS 30m land cover raster to each ecozone boundary.
  2. Reclassifying each ecotype cell to its ecozone-specific score using the provided scores table.

Inputs

  • data/Ecozones/ecozones.shp: Ecozone shapefile
  • data/can_land_cover_2020v2_30m_tif/CAN_NALCMS_landcover_2020v2_30m/data/CAN_NALCMS_landcover_2020v2_30m.tif: Ecotype raster from NALCMS
  • outputs/Ecotype_Scores_Scaled_Final.csv: Ecotype scores (metadata)

9.1.1 Load Libraries

library(terra)
library(sf)
library(dplyr)

9.2 Ecozone Rasters

9.2.1 Read Files and Prepare Data

# Load ecozones shapefile and ecotypes raster
ecozones <- st_read("data/Ecozones/ecozones.shp")
ecotypes <- rast("data/can_land_cover_2020v2_30m_tif/CAN_NALCMS_landcover_2020v2_30m/data/CAN_NALCMS_landcover_2020v2_30m.tif")
scores <- read.csv("outputs/Ecotype_Scores_Scaled_Final.csv")
# Reproject ecozones to match raster CRS
ecozones <- st_transform(ecozones, crs(ecotypes))

# Create habitat lookup table
habitat_lookup <- data.frame(
  value = 1:19,
  name = c(
    "Temperate or sub-polar needleleaf forest", "Sub-polar taiga needleleaf forest",
    "Tropical or sub-tropical broadleaf evergreen forest", "Tropical or sub-tropical broadleaf deciduous forest",
    "Temperate or sub-polar broadleaf deciduous forest", "Mixed Forest",
    "Tropical or sub-tropical shrubland", "Temperate or sub-polar shrubland",
    "Tropical or sub-tropical grassland", "Temperate or sub-polar grassland",
    "Sub-polar or polar shrubland-lichen-moss", "Sub-polar or polar grassland-lichen-moss",
    "Sub-polar or polar barren-lichen-moss", "Wetland", "Cropland", "Barren lands",
    "Urban", "Water", "Snow and Ice"
  )
)

# Get all unique ecozone names
all_zones <- unique(ecozones$ZONE_NAME)

# Set the output directory
output_dir <- "outputs/ecozone_rasters"
dir.create(output_dir, showWarnings = FALSE)

9.2.2 Ecozone Masking Loop

This loop creates an independent raster for each ecozone, where each ecotype is reclassified to its respective score for that ecozone.

## Ecozone Loop
for (zone in all_zones) {
  cat("Processing:", zone, "\n")
  try({
    # Filter to current ecozone polygon
    zone_poly <- ecozones %>%
      filter(ZONE_NAME == zone) %>%
      st_make_valid()
    
    if (nrow(zone_poly) == 0 || st_is_empty(zone_poly)) {
      cat("No geometry found for", zone, "- skipping.\n\n")
      next
    }
    
    zone_vect <- vect(zone_poly)
    
    # Crop and mask raster
    eco_crop <- terra::crop(ecotypes, zone_vect)
    eco_masked <- terra::mask(eco_crop, zone_vect)
    
    # Remove "Snow and Ice" (value 19)
    eco_masked[eco_masked == 19] <- NA
    
    # Filter scores for current ecozone
    zone_scores <- scores %>% filter(Ecozone == zone)
    
    # Merge with the habitat lookup
    score_lookup <- merge(
      habitat_lookup, zone_scores,
      by.x = "name", by.y = "Ecotype", all.x = TRUE
    )
    score_lookup_clean <- score_lookup[!is.na(score_lookup$Score), ]
    
    # If no valid scores, skip
    if (nrow(score_lookup_clean) == 0) {
      cat("⚠ No matching ecotype scores for", zone, "- skipping.\n\n")
      next
    }
    
    # Reclassify to corresponding scores
    reclass_matrix <- as.matrix(score_lookup_clean[, c("value", "Score")])
    score_raster <- classify(eco_masked, rcl = reclass_matrix, others = NA)
    
    # Save as .rds
    rds_path <- file.path(output_dir, paste0(gsub(" ", "_", zone), "_score_raster.rds"))
    saveRDS(score_raster, rds_path)
    
    # Save as .tif (GeoTIFF)
    tif_path <- file.path(output_dir, paste0(gsub(" ", "_", zone), "_score_raster.tif"))
    writeRaster(score_raster, filename = tif_path, overwrite = TRUE)
    
    cat("✔ Finished:", zone, "\n\n")
  }, silent = FALSE)
}

9.4 Overview: Ecotype Loop

The Ecotype Loop extracts key ecotypes across Canada and outputs separate rasters for each

The workflow includes:

  1. Identify target ecotype scores from the CSV table.
  2. Reclassify all non-target scores to NA.

Inputs

  • outputs/ecozone_rasters/mosaic/Canada_Ecozone_Score_Mosaic_FIRST_15.tif: mosaicked national ecotype raster
  • outputs/Ecotype_Scores_Scaled_Final.csv: ecotype metadata

9.4.1 Load Libraries

library(terra)
library(dplyr)

9.5 Ecotype Rasters

9.5.1 Data Preparation and File Loading

# Load raster and scores table
r <- rast("outputs/ecozone_rasters/mosaic/Canada_Ecozone_Score_Mosaic_FIRST_15.tif")
scores <- read.csv("outputs/Ecotype_Scores_Scaled_Final.csv")

# Output folder
out_dir <- "outputs/Reclassified_By_Ecotype/"

# List of ecotypes to process
ecotypes <- c(
  "Barren lands",
  "Mixed Forest",
  "Temperate or sub-polar grassland",
  "Temperate or sub-polar broadleaf deciduous forest",
  "Temperate or sub-polar needleleaf forest",
  "Water",
  "Wetland"
)

# Tolerance for matching scores
tolerance <- 0.001

9.5.2 Loop Through Ecotypes

# Loop through ecotypes
for (eco in ecotypes) {

  # Filter and extract target scores
  target_scores <- unique(scores %>% filter(Ecotype == eco) %>% pull(Score))
  
  # Skip if no matching scores
  if (length(target_scores) == 0) {
    warning(paste("No scores found for:", eco))
    next
  }
  
  # Build reclass matrix
  reclass_matrix <- do.call(rbind, lapply(target_scores, function(x) {
    c(x - tolerance, x + tolerance, x)
  }))
  
  # Reclassify
  r_reclassed <- classify(r, rcl = reclass_matrix, others = NA)
  
  # Clean ecotype name for file output
  clean_name <- gsub("[^a-zA-Z0-9]", "_", eco)
  
  # Save as .tif
  writeRaster(
    r_reclassed,
    filename = file.path(out_dir, paste0("Reclassified_", clean_name, ".tif")),
    overwrite = TRUE,
    wopt = list(datatype = "FLT4S", gdal = c("COMPRESS=LZW", "TILED=YES"))
  )
  
  # Save as .rds
  saveRDS(r_reclassed, file = file.path(out_dir, paste0("Reclassified_", clean_name, ".rds")))
}
